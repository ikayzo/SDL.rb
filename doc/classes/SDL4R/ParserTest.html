     # File test/sdl4r/parser_test.rb, line 388
388:     def test_parse_error
389:       # WARNING: the line and col of an error is not accurate science. The goal here is to point to
390:       # coordinates that are useful to the user.
391:       # Exampe for a string litteral that spans over several line, some errors could be point to
392:       # the start or to the end without too much ambiguity.
393:       # Consequently, feel free to change the coordinates, if a change in the implementation
394:       # modifies the x/y of the error and they still make sense.
395:       assert_error_xy "=", 1, 1
396:       assert_error_xy "tag1 xyz", 1, 6
397:       assert_error_xy "tag1 \\\nxyz", 2, 1
398:       assert_error_xy "tag1 \\\n   xyz", 2, 4
399:       
400:       source = "-- my comment\n=\n"
401:       assert_error_xy source, 2, 1
402: 
403:       source = "murder_plot 123 \\\\\n      weight=456 \\\\\n      * \\\\\n      length=789\n"
404:       assert_error_xy source, 3, 6
405: 
406:       assert_error_xy 'tag1 "text\\"', 1, 13
407: 
408:       source = "murder_plot \"abcd\\\\\n      efghij\\\\\n      kl\\\\\n      mnopq\n"
409:       assert_error_xy source, 4, 13
410:     end
